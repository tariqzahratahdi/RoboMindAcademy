# to let Robo do a slalom under

# Challenge: Understand what is
# repeated and what is unique
# in this kind of taks.

# Solution:

# How to go to the start?
forward(6)
right
forward(3)

# Repeat this how many times?
repeat(3)
{
	forward(2)
    right
    forward(2)
    left
    forward(2)
    left
    forward(2)
    right
}
