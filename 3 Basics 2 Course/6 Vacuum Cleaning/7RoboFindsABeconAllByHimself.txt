# Description: Let the robot find
# a beacon at an unknown location
# without help of the environment.

# Challenge: Use flipCoin in combination
# with repeat{...} as a search strategy.

# Solution:

repeat
{
	if(frontIsBeacon)
	{
		# 1. Complete:
		eatUp
		end
	}

	if (flipCoin)
	{
		# 2. Complete:
		right
		forward(1)

	}
	else
	{
		# 3. Complete:
		left
		forward(1)
	}
}
