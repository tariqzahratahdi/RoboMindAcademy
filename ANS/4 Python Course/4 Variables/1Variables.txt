# Challenge:  Use a variable to draw lines with a decreasing length. Start with length 3.

#Solution

def paint_line(l):
    robo.paintWhite()
    robo.forward(l)
    robo.stopPainting()
    robo.backward(l)


length = 3
while length > 0:
    paint_line(length)
    robo.right()
    robo.forward()
    robo.left()
    length = length - 1
