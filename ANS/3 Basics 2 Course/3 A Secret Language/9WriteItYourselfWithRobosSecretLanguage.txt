# Description: Write it yourself
# with Robo's secret language.

# Challenge: use procedures.

# Solution:

# Write ALP:
secretA
secretL
secretP

procedure secretL
{
	paintWhite
	stopPainting
	right
	forward(1)
	left
}

procedure secretP
{
	paintBlack
	forward(2)
	stopPainting
	backward(2)
	right
	forward(1)
	left
}

procedure secretA
{
	paintWhite
	repeat(4)
	{
		forward(2)
		right
	}
	stopPainting
	right
	forward(3)
	left
}
