# Description: Let the robot
# follow the white line to find
# the beacon and pick it up.

# Challenge: Use repeat(.){...}
# if(.){...} and tests.

# Solution:

repeat
{
	if(frontIsWhite)
	{
		# 1. Complete:
		forward(1)
	}
	
	if(rightIsWhite)
	{
		# 2. Complete:
		right
	}
	if(leftIsWhite)
	{
		# 3. Complete:
		left
	}

	if(frontIsBeacon)
	{
		# 4. Complete:
		pickUp
		break
	}
}
